# Physical

departments
-
dept_no VARCHAR(255) PK 
dept_name VARCHAR(255)

dept_emp
-
emp_no INT PK
dept_no VARCHAR(255) FK - departments.dept_no

dept_manager
-
dept_no VARCHAR(255) FK - departments.dept_no
emp_no INT FK - dept_emp.emp_no

employees
-
emp_no INT FK - dept_manager.emp_no
emp_title_id VARCHAR(255) FK - titles.title_id
birth_date VARCHAR(255)
first_name VARCHAR(255)
last_name VARCHAR(255)
sex VARCHAR(1)
hire_date VARCHAR(255)

salaries
-
emp_no VARCHAR(255) FK - dept_emp.emp_no
salary INT 

titles
-
title_id VARCHAR(255) PK
title VARCHAR(255)
